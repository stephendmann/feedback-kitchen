**Replace with:**

```js
function _aiGuards() {
  if (!scoreResult || !scoreResult.rows.some(r => r.grade)) {
    alert('Grade at least one criterion before using AI Assist.');
    return false;
  }
  const weightSum = scoreResult.rows
    .filter(function (r) { return r.grade; })
    .reduce(function (s, r) { return s + r.criterion.weight; }, 0);
  const displayWeightSum = Number.isInteger(weightSum) ? weightSum : parseFloat(weightSum.toFixed(2));
  // Over-100% is a real configuration error — keep as a hard block.
  if (Math.round(weightSum) > 100) {
    alert('Graded criteria weight ' + displayWeightSum + '% — exceeds 100%. Check your weights before using AI Assist.');
    return false;
  }
  // Under-100% is a legitimate partial-mark scenario — soft confirm instead.
  if (Math.round(weightSum) < 100) {
    const proceed = confirm(
      'Only ' + displayWeightSum + '% of criteria are graded.\n\n' +
      'AI Assist will generate feedback only for the criteria you have graded. The TOTAL SCORE line will reflect ' + displayWeightSum + ' out of 100, not a full 100-point mark.\n\n' +
      'Continue?'
    );
    if (!proceed) return false;
  }
  return true;
}
```

**Behaviour after change:**

- 0% graded → hard block (unchanged).
- 1–99% graded → confirm dialog; proceed if user OKs.
- Exactly 100% graded → no dialog, straight through (unchanged).
- Over 100% graded → hard block (misconfigured weights).
