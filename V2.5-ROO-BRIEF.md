## 2.5-1 — Keyboard shortcuts

**File:** `scorer.html`

**Why not `Ctrl+C` / `Ctrl+E`?** `Ctrl+C` is copy-selected-text; overloading it breaks user expectations. `Ctrl+E` focuses the browser search bar in Chrome and Firefox. Both are poor choices. Use `Ctrl+Shift+…` combinations instead — rare enough to be safe, still easy to hit one-handed.

**Bindings:**

| Shortcut | Action | Notes |
| --- | --- | --- |
| `Ctrl+Shift+C` / `Cmd+Shift+C` | `S.copyFeedback()` | Same as sticky bar button |
| `Ctrl+Shift+E` / `Cmd+Shift+E` | `S.downloadExcel()` | Single-student Marker's Record |
| `Ctrl+Shift+X` / `Cmd+Shift+X` | `S.exportCohort()` | Cohort workbook |
| `Ctrl+Shift+N` / `Cmd+Shift+N` | `S.newStudent()` | Confirms first via existing flow |
| `?` (plain Shift+/) | Open shortcuts modal | Only when no input/textarea focused |
| `Esc` | Close any open modal | Cohort list, cohort setup, AI settings, shortcuts |

Add inside the scorer `<script>` at the bottom of the IIFE (before the `return { … };` block):

```js
function bindKeyboardShortcuts() {
  function isEditing() {
    const t = document.activeElement;
    if (!t) return false;
    const tag = t.tagName;
    return tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT' || t.isContentEditable;
  }

  function closeAnyOpenModal() {
    ['cohort-list-modal', 'cohort-setup-modal', 'cohort-wipe-modal', 'ai-settings-modal', 'shortcuts-modal']
      .forEach(function (id) {
        const m = document.getElementById(id);
        if (m && !m.classList.contains('hidden')) m.classList.add('hidden');
      });
  }

  document.addEventListener('keydown', function (e) {
    // Esc — always close modals
    if (e.key === 'Escape') { closeAnyOpenModal(); return; }

    // ? — open shortcuts modal (only if not editing)
    if (e.key === '?' && !e.ctrlKey && !e.metaKey && !isEditing()) {
      e.preventDefault();
      const m = document.getElementById('shortcuts-modal');
      if (m) m.classList.remove('hidden');
      return;
    }

    // Ctrl/Cmd + Shift + <letter>
    const mod = (e.ctrlKey || e.metaKey) && e.shiftKey;
    if (!mod) return;
    const key = e.key.toLowerCase();

    if (key === 'c') { e.preventDefault(); copyFeedback(); return; }
    if (key === 'e') { e.preventDefault(); downloadExcel(); return; }
    if (key === 'x') { e.preventDefault(); exportCohort(); return; }
    if (key === 'n') { e.preventDefault(); newStudent(); return; }
  });
}
```

Call `bindKeyboardShortcuts();` at the end of `init()`.

**Acceptance:**

- Each shortcut fires the same function as its sticky-bar or nav button.
- `?` opens the shortcut modal only when not inside an input/textarea (so typing `?` in feedback text doesn't hijack).
- `Esc` closes any visible modal.
- No browser default is visibly shadowed (Ctrl+Shift+C still works for DevTools *when DevTools is already open and focused* — acceptable trade-off).

## 2.5-4 — Persist section open/closed state to localStorage

**File:** `scorer.html`

Each `<details>` gets its open/closed state remembered across reloads and across students. Add these helpers inside the IIFE:

```js
const SECTION_STATE_KEY = 'SA_SECTION_STATE_V1';

function persistSectionState() {
  const state = {};
  document.querySelectorAll('main details, section details').forEach(function (d) {
    if (!d.id) return;
    state[d.id] = !!d.open;
  });
  try { localStorage.setItem(SECTION_STATE_KEY, JSON.stringify(state)); } catch (e) {}
}

function restoreSectionState() {
  let state = {};
  try { state = JSON.parse(localStorage.getItem(SECTION_STATE_KEY) || '{}'); } catch (e) {}
  document.querySelectorAll('main details, section details').forEach(function (d) {
    if (!d.id) return;
    if (d.id in state) d.open = !!state[d.id];
  });
}

function wireSectionStatePersistence() {
  document.querySelectorAll('main details, section details').forEach(function (d) {
    if (!d.id) return;
    d.addEventListener('toggle', persistSectionState);
  });
}
```

In `init()`, call `restoreSectionState()` **before** the AI panel auto-open logic runs, and call `wireSectionStatePersistence()` after the DOM is fully rendered.

Export `persistSectionState` and `restoreSectionState` in the `return { … }` block.

**Edge cases to handle:**

- Ensure every `<details>` in sections E, F, G, H has a stable `id` attribute (Section F already does: `ai-garnish-panel`). Add `id="sec-notes-details"`, `id="sec-ai-details"` (if different from the outer section id), etc., wherever needed to avoid collisions with the anchor-target IDs.
- If a brand-new user has no `SA_SECTION_STATE_V1` key, keep the v2.2 defaults (A/B/C/D open, E/F closed) by not overriding unset keys.
- The AI auto-open on successful `aiAssist()` call must still work — after opening, it triggers `toggle`, which saves the new state. That's the desired behaviour.

**Acceptance:** expand F, reload page → F is still open. Collapse B, reload → B is still collapsed. Export all collapsed, reload → all collapsed.
